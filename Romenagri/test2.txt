laa _imakana _ulakzaula ba_aana _ula_hzaasauba _hau _ukhza_taraa_aee ba_hza_da jzzaa_ta_ha, la_aana_ha kaana na_taaja _ulakaszza_ira mana _ulaaba_takaaraa_ta _ula_aeelamayyai waala_ta_tzaba_ikzaa_ta _ularayyaa_jzayyai. _ula_hzawaasa_iba ma_tanauyyai phza_i _ulawaakza_aee, wa_tzabakzaa_m laphzara_jza _tasharasha phza_i _aalyai _tauranagza phza_aana _hzaasaubaa_m la_ha kza_dara_ai jzzaa_ta _hza_da manakhzaphza_jza _ikauna kzaa_daraa_m _aeela_aa _aanajaajza _ulama_haama _ulakhzaasza_ai ba_aa_i _hzaasauba _aakhzara, ba_da_haa_m mana _ulamasaa_aee_da _ularakzama_i _ulashakhzasza_i _aala_aa _ula_hzaasauba _ulaphzaayakza, _tzaalamaa _aana _ulaukza_ta wasayyai _ulajzzaakara_ai la_isa_ta phza_i _ulaa_aee_tabaara. lajzzalaka phza_aana _ula_taszama_imaa_ta _ulama_tamaaszzalyai mana _ula_hzaasauba mana _ulamamakana _aana _ta_jzaba_tza mana _aajala ma_haama _ta_tara_ou_hza ba_ina mayaalajyai _hzasaabaa_ta mau_jzzaphza_i _ulasharakaa_ta waala_ta_hzakama phza_i _ulamarakabaa_ta _ulaphza_jzaayayyai ba_dauna _tzayyaara. wabasababa _ula_ta_tzaura _ula_takzana_i phza_aana _ula_hzawaasa_iba _ula_hza_da_iszza_ai _takauna bashakala jabara_i _aakaszzara kza_dara_ai mana _talaka _ula_ta_i mana _ula_aajayyaala _ulasaabakza_ai wa_ha_i _jzzaa_hara_ai mauszauphza_ai wamasharau_hza_ai jajzayayyaa phza_i kzaanauna maura.

_aaszala _ula_tasamayyai
_aa_tzalakza shaarala baabaaja kalama_ai ""computer"" _aeela_aa _ulashakhzasza _ulajzza_i _i_dakhzala _ulabayyaanaa_ta _aala_aa _ula_hzaasauba, waphza_imaa ba_aee_da _aa_tzalakza_ta _ulakalama_ai _aeela_aa _ula_aalyai naphzasa_haa. phza_imaa _aayauja_da_ta kalama_ai <<_hzaasauba>> ka_tarajama_ai lalaphza_jzza_ai _ula_aanajala_ijzayyai ""computer"", wajzzalaka la_aana _hajzza_ha _ulakalama_ai masha_takza_ai mana _ulaphza_aeela "e_compute" _ula_ta_i _ta_aeena_i _i_hzasaba wa_i_jzaaphza phza_i _ulalagza_ai _ula_aanakala_ijzayyai waala_hzaraphza_ina "e_er" phza_i _aakhzara ba_aee_jza _ula_aaphzayaala la_ta_hzaula_haa _aala_aa _usama phzaa_aeela phza_taszaba_hza _hzaasaba _aayau _hzaasauba. wajzzalaka la_isa gzara_ibaa_m la_aana _ula_hzaasauba mabana_i _aeela_aa _aana _ikzauma bajama_i_aee _aeemalayyaa_ta_ha _aeela_aa shakala _hzasaabaa_ta: jama_aee _aayau _tzara_hza _aayau _jzaraba _aayau kzasama_ai baasa_takhza_daama na_jzzaama _ula_aee_da _ulaszzanaaya_i. maszzaala la_aeera_jza szaura_ai: _ikzauma ba_tajajza_i _ulamasa_aalyai la_aajajzaa _aaszagzara wa_ha_i _ulanakzaa_tza ("e_pixel"), kala nakza_tza_ai _ha_i _aeebaara_ai _aeena launa wamau_jza_aee _hajzza_ha _ulanakza_tza_ai baalanasaba_ai lalaszaura_ai, _hau ma_hzaszalyai szzalaaszza_ai _aalawaana: _ula_aajzarakza, _ula_aakhza_jzara waala_aa_hzamara, kala launa _imakana _tamaszza_ila _daraja_ta_ai bakza_ima_ai (_ta_tara_ou_hza ba_ina 0 wa255 maszzalaa_m), _aajzzaa_m _ulalauna _hau rakzama baalanasaba_ai lala_hzaasaba (maszzalaa_m _ula_aaba_i_jza: 255,255,255) wa_aeena_damaa _ira_i_da _aa_jzza_haara _hajzzaa _ulalauna _irasala _ulakza_ima_ai _ulamawaaphzakza_ai laba_tzaakza_ai _ularasaumayyaa_ta. wa_aaszala kalama_ai _hzaasauba phza_i _ulalagza_ai mana _ulamasza_dara _hzasaba, _aeela_aa wajzana _usama _ula_aalyai <<phphzaa_aee_uyoala>>.
